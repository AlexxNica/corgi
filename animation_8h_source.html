38 /// @struct AnimationData
39 ///
40 /// @brief The Component data structure that corresponds to each Entity that
41 /// is registered with the AnimationComponent. It contains all the information
42 /// for each Entity's animation.
43 struct AnimationData {
44  AnimationData()
45  : anim_table_object(-1),
46  last_anim_idx(-1),
47  previous_time_remaining(motive::kMotiveTimeEndless),
48  debug_state(AnimationDebugState_None),
49  debug_bone(motive::kInvalidBoneIdx) {}
50 
51  /// @var motivator
52  ///
53  /// @brief Holds and processes the animaton. Calls can be made to
54  /// `motivator.GlobalTransforms()` to get an array of matrices: one for
55  /// each bone in the animation.
56  motive::RigMotivator motivator;
57 
58  /// @var anim_table_object
59  ///
60  /// @brief Specifies the object-type query parameter for the AnimTable,
61  /// which is used to index the first array of the AnimTable.
62  int anim_table_object;
63 
64  /// @var last_anim_idx
65  ///
66  /// @brief Tracks the index of the last animation that was played by
67  /// `AnimateFromTable()`.
68  int last_anim_idx;
69 
70  /// @var previous_time_remaining
71  ///
72  /// @brief The amount of time remaning from the previous animation. This
73  /// can be used for firing animation events.
74  motive::MotiveTime previous_time_remaining;
75 
76  /// @var debug_state
77  ///
78  /// @brief The debug information output to the log every frame, for this
79  /// animating object. You probably want this to be None for all but one
80  /// animating object, since it outputs a lot of data.
81  AnimationDebugState debug_state;
82 
83  /// @var debug_bone
84  ///
85  /// @brief When `debug_state` outputs bone-specific information, this
86  /// member specifies the bone. Ignored otherwise.
87  motive::BoneIndex debug_bone;
88 };

90 /// @class AnimationComponent
91 ///
92 /// @brief A Component that provides an elegant way to handle Entity animation
93 /// by interacting with the Motive animation library.
94 class AnimationComponent : public Component<AnimationData> {
95  public:
96  /// @brief Deconstructor of the Animation component.
97  virtual ~AnimationComponent() {}
98 
99  /// @brief Updates all Motivators in the AnimationData, as well as, any other
100  /// Motivators that were initialized with AnimationComponent's MotiveEngine.
101  ///
102  /// @param[in] delta_time An corgi::WorldTime representing the delta time
103  /// since the last call to UpdateAllEntities.
104  virtual void UpdateAllEntities(corgi::WorldTime delta_time);
105 
106  /// @brief Deserialize a flat binary buffer to create and populate an Entity
107  /// from raw data.
108  ///
109  /// @param[in,out] entity An EntityRef reference that points to an Entity that
110  /// is being added from the raw data.
111  /// @param[in] data A void pointer to the raw FlatBuffer data.
112  virtual void AddFromRawData(corgi::EntityRef& entity, const void* data);
113 
114  /// @brief Serializes an AnimationComponent's data for a given Entity.
115  ///
116  /// @param[in] entity An EntityRef reference to an Entity whose corresponding
117  /// AnimationData will be serialized.
118  ///
119  /// @return Returns a RawDataUniquePtr to the start of the raw data in a
120  /// flat binary buffer.
121  virtual RawDataUniquePtr ExportRawData(const corgi::EntityRef& entity) const;
122 
123  /// @brief Begin playback of a given RigAnim on a given Entity.
124  ///
125  /// @param[in] entity A const EntityRef reference to an Entity that the
126  /// RigAnim `anim` should be applied on.
127  ///
128  /// @warning This method asserts that `entity` must also have a
129  /// RenderMeshComponent in order for the animation to be applied.
130  ///
131  /// @param[in] anim A const motive::RigAnim reference to the animation that
132  /// should be applied on `entity`.
133  void Animate(const corgi::EntityRef& entity, const motive::RigAnim& anim);
134 
135  /// @brief Query the AnimTable for an animation and then play it.
136  ///
137  /// @param[in] entity A const EntityRef reference to the Entity whose
138  /// corresponding animation should be started.
139  /// @param[in] anim_idx An int index of the animation that should be
140  /// started.
141  ///
142  /// @return Returns `true` if a new animation was successfully started.
143  /// Otherwise, it returns `false` if the animation could not be found in
144  /// the AnimTable.
145  bool AnimateFromTable(const corgi::EntityRef& entity, int anim_idx);
146 
147  /// @brief Check if a certain animation exists with a given index in the
148  /// AnimTable for a given Entity.
149  ///
150  /// @param[in] entity A const EntityRef reference to the Entity whose
151  /// corresponding animation should be checked.
152  /// @param[in] anim_idx An int index of the animation whose existence should
153  /// be checked.
154  ///
155  /// @return Returns `true` if the animation exists in the AnimTable with
156  /// `anim_idx`. Returns `false` if the animation is not found in the
157  /// AnimTable.
158  bool HasAnim(const corgi::EntityRef& entity, int anim_idx) const {
159  const AnimationData* data = Data<AnimationData>(entity);
160  return anim_table_.Query(data->anim_table_object, anim_idx) != nullptr;
161  }
162 
163  /// @brief Get the length of an Entity's animation, given an animation index.
164  ///
165  /// @param[in] entity A const EntityRef reference to the Entity whose
166  /// corresponding animation's length should be returned.
167  /// @param[in] anim_idx An int index of the animation whose length should be
168  /// returned.
169  ///
170  /// @return Returns the length of the animation, if it exists. Otherwise, it
171  /// returns 0.
172  motive::MotiveTime AnimLength(const corgi::EntityRef& entity,
173  int anim_idx) const {
174  const AnimationData* data = Data<AnimationData>(entity);
175  const motive::RigAnim* anim =
176  anim_table_.Query(data->anim_table_object, anim_idx);
177  return anim == nullptr ? 0 : anim->end_time();
178  }
179 
180  /// @brief Get the index of the last animation that was played.
181  ///
182  /// @param[in] entity A const EntityRef reference to the entity whose index
183  /// should be returned.
184  ///
185  /// @return Returns the index of the last animation that was played via
186  /// `AnimateFromTable()`, or returns -1 if no animation has ever been played
187  /// using that function.
188  int LastAnimIdx(const corgi::EntityRef& entity) const {
189  return Data<AnimationData>(entity)->last_anim_idx;
190  }
191 
192  /// @brief Get a reference to the engine that can be used for external
193  /// Motivators as well. For the greatest efficiency, there should only be one
194  /// MotiveEngine.
195  ///
196  /// @return Returns a reference to the MotiveEngine for this
197  /// AnimationComponent.
198  motive::MotiveEngine& engine() { return engine_; }
199 
200  /// @brief Get a const reference to the engine that can be used for
201  /// external Motivators as well. For the greatest efficiency,
202  /// there should only be one MotiveEngine.
203  ///
204  /// @return Returns a const reference to the MotiveEngine for this
205  /// AnimationComponent.
206  const motive::MotiveEngine& engine() const { return engine_; }
207 
208  /// @brief Get a reference to the animation table that is queried when
209  /// `AnimateFromTable()` is called.
210  ///
211  /// @return Returns a reference to the AnimTable for this AnimationComponent.
212  motive::AnimTable& anim_table() { return anim_table_; }
213 
214  /// @brief Get a const reference to the animation table that is queried when
215  /// `AnimateFromTable()` is called.
216  ///
217  /// @return Returns a const reference to the AnimTable for this
218  /// AnimationComponent.
219  const motive::AnimTable& anim_table() const { return anim_table_; }
220 
221  private:
222  /// @brief Sets the motivator up for any of the animations on the `object`.
223  void InitializeMotivator(const corgi::EntityRef& entity);
224 
225  /// @var engine_
226  ///
227  /// @brief Holds MotiveProcessors that, in turn, hold the animation state.
228  /// Calling AdvanceFrame() on this updates all the animations at once.
229  motive::MotiveEngine engine_;
230 
231  /// @var anim_table_
232  ///
233  /// @brief Holds a table of animations.
234  /// It is a simple array of arrays. The top array is indexed by
235  /// `anim_table_object` in `AnimationData`. The bottom array is indexed by
236  /// `anim_idx` in the `AnimateFromTable()` call.
237  motive::AnimTable anim_table_;
238 };
239 /// @}
