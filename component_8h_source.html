31 /// @class Component
32 /// @brief A Component is an object that encapsulates all data and logic
33 /// for Entities of a particular type.
34 ///
35 /// @note On some of the API, EntityRef& parameters are non-const so that
36 /// the referenced Entity can be changed. The EntityRef itself is never
37 /// modified.
38 ///
39 /// @tparam T The structure of data that needs to be associated with each
40 /// Entity.
41 template <typename T>
42 class Component : public ComponentInterface {
43  public:
44  /// @struct ComponentData
45  /// @brief A structure of data that is associated with each Entity.
46  ///
47  /// It contains the template struct, as well as a pointer back to the
48  /// Entity that owns this data.
49  struct ComponentData {
50  /// @brief The default constructor for an empty ComponentData.
51  ComponentData() {}
52 
53  /// @var entity
54  ///
55  /// @brief The Entity associated with this data.
56  EntityRef entity;
57 
58  /// @var data
59  ///
60  /// @brief The data to associate with the Entity.
61  T data;
62 
63  /// @brief Construct a new ComponentData from an existing ComponentData.
64  ///
65  /// @param[in] src An existing ComponentData whose data should be moved
66  /// into the new ComponentData.
67  ComponentData(ComponentData&& src) {
68  entity = std::move(src.entity);
69  data = std::move(src.data);
70  }
71 
72  /// @brief Move a referenced ComponentData into this ComponentData.
73  ///
74  /// Move operators are efficient since they allow referenced data to
75  /// be moved instead of copied.
76  ///
77  /// @param[in] src A referenced ComponentData to be moved into this
78  /// ComponentData.
79  ComponentData& operator=(ComponentData&& src) {
80  entity = std::move(src.entity);
81  data = std::move(src.data);
82  return *this;
83  }
84 
85  private:
86  ComponentData(const ComponentData&);
87  ComponentData& operator=(const ComponentData&);
88  };

107  /// @brief Provides an alternate way to add Entities if you do not

110  ///
111  /// @note AddEntity is a much better function, except we cannot have it in

118  /// @brief Adds an Entity to the list that this Component is tracking.
119  ///
120  /// @param[in,out] entity An EntityRef reference used to add an Entity to the
121  /// list of Entities that this Component keeps track of.
122  /// @param[in] alloc_location An Enum that specifies whether to allocate from
123  /// the beginning or end of the memory pool.
124  ///
125  /// @return Returns the data structure associated with the Component.
126  ///
127  /// @note If you have already registered for this Component, this
128  /// will just return a reference to the existing data and will not change
129  /// anything.
130  T* AddEntity(EntityRef& entity, AllocationLocation alloc_location) {
131  if (entity->IsRegisteredForComponent(GetComponentId())) {
132  return GetComponentData(entity);
133  }
134  // No existing data, so we allocate some and return it:
135  ComponentIndex index = static_cast<ComponentIndex>(
136  component_data_.GetNewElement(alloc_location).index());
137  entity->SetComponentDataIndex(GetComponentId(), index);
138  ComponentData* component_data = component_data_.GetElementData(index);
139  component_data->entity = entity;
140  InitEntity(entity);
141  return &(component_data->data);
142  }

144  /// @brief Adds an Entity to the list that this Component is tracking.
145  ///
146  /// @note Entities added through this function allocate from the back of the
147  /// memory pool.
148  ///
149  /// @param[in,out] entity An EntityRef reference used to add an Entity to the
150  /// list of Entities that this Component keeps track of.
151  ///
152  /// @return Returns the data structure associated with the Component.
153  T* AddEntity(EntityRef& entity) { return AddEntity(entity, kAddToBack); }

155  /// @brief Removes an Entity from the list of Entities.
156  ///
157  /// This is done by marking the Entity as no longer using this Component,
158  /// calling the destructor on the data, and returning the memory to the
159  /// memory pool.
160  ///
161  /// @param[in,out] entity An EntityRef reference used to remove an Entity
162  /// from the list of Entities that this Component keeps track of.
163  virtual void RemoveEntity(EntityRef& entity) {
164  RemoveEntityInternal(entity);
165  component_data_.FreeElement(GetComponentDataIndex(entity));
166  entity->SetComponentDataIndex(GetComponentId(), kUnusedComponentIndex);
167  }
168 
169  /// @brief Removes an Entity from the list of Entities.
170  ///
171  /// This is done by marking the Entity as no longer using this Component,
172  /// calling the destructor on the data, and returning the memory to the
173  /// memory pool.
174  ///
175  /// @param[in,out] iter An EntityIterator that references an Entity that
176  /// should
177  /// be removed.
178  ///
179  /// @return Returns an iterator to the next Entity after the one that was
180  /// just removed.
181  virtual EntityIterator RemoveEntity(EntityIterator iter) {
182  EntityRef entity = iter->entity;
183  RemoveEntityInternal(entity);
184  EntityIterator new_iter = component_data_.FreeElement(iter);
185  entity->SetComponentDataIndex(GetComponentId(), kUnusedComponentIndex);
186  return new_iter;
187  }

244  /// @brief Gets the Component data stored at a given index.
245  ///
246  /// @warning This pointer is NOT stable in memory. Calls to AddEntity and
247  /// AddEntityGenerically may force the storage class to resize,
248  /// shuffling around the location of this data.
249  ///
250  /// @param[in] data_index A size_t representing the index of the desired
251  /// Component data.
252  ///
253  /// @return Returns a pointer of the data structure associated with the
254  /// Component data, or returns a nullptr if given an invalid data_index.
255  T* GetComponentData(size_t data_index) {
256  if (data_index == kUnusedComponentIndex) {
257  return nullptr;
258  }
259  ComponentData* element_data = component_data_.GetElementData(data_index);
260  return (element_data != nullptr) ? &(element_data->data) : nullptr;
261  }

263  /// @brief Gets the data for a given Entity.
264  ///
265  /// @warning This pointer is NOT stable in memory. Calls to AddEntity and
266  /// AddEntityGenerically may force the storage class to resize,
267  /// shuffling around the location of this data.
268  ///
269  /// @param[in] entity An EntityRef reference to the Entity whose data should
270  /// be returned.
271  ///
272  /// @return Returns the Entity's data as a pointer of the data structure
273  /// associated with the Component data, or returns a nullptr if the data
274  /// does not exist.
275  T* GetComponentData(const EntityRef& entity) {
276  size_t data_index = GetComponentDataIndex(entity);
