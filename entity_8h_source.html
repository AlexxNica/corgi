22 /// @file

26 /// @class Entity
27 ///
28 /// @brief The basic Entity class for CORGI. It contains an array
29 /// of index values, which are used by Components for tracking their data
30 /// associated with an Entity. It also contains a boolean for tracking if
31 /// this Entity has been marked for deletion.
32 class Entity {
33  public:
34  /// @brief Constructor for creating an Entity.
35  Entity() : marked_for_deletion_(false) {
36  for (int i = 0; i < kMaxComponentCount; i++) {
37  componentDataIndex_[i] = kUnusedComponentIndex;
38  }
39  }
40 
41  /// @brief Gets the index of the data in the corresponding Component
42  /// system.
43  ///
44  /// @param[in] componentId The ID of the Component whose data
45  /// index will be returned.
46  ///
47  /// @return Returns an int index of the data in the corresponding
48  /// Component system.
49  int GetComponentDataIndex(ComponentId componentId) const {
50  return componentDataIndex_[componentId];
51  }
52 
53  /// @brief Sets the index for the data associated with this Entity in the
54  /// given Component.
55  ///
56  /// @param[in] componentId The ID of the Component whose data index will
57  /// be set.
58  /// @param[in] value The ComponentIndex value to set for the given Component.
59  void SetComponentDataIndex(ComponentId componentId, ComponentIndex value) {
60  componentDataIndex_[componentId] = value;
61  }
62 
63  /// @brief A utility function for checking if this Entity is associated
64  /// with a given Component.
65  ///
66  /// @param[in] componentId The ID of the Component to check whether or not
67  /// it is associated with this Entity.
68  ///
69  /// @return Returns a bool of whether or not this Entity is associated
70  /// with the given Component.
71  bool IsRegisteredForComponent(ComponentId componentId) const {
72  return componentDataIndex_[componentId] != kUnusedComponentIndex;
73  }
74 
75  /// @brief An accessor function to check if this Entity has been
76  /// marked for deletion.
77  ///
78  /// @return Returns a bool indicating whether or not this Entity
79  /// is marked for deletion.
80  bool marked_for_deletion() const { return marked_for_deletion_; }
81 
82  /// @brief A mutator function to set an Entity as marked for deletion.
83  ///
84  /// @param[in] marked_for_deletion A bool corresponding to whether the Entity
85  /// should be marked for deletion or not.
86  void set_marked_for_deletion(bool marked_for_deletion) {
87  marked_for_deletion_ = marked_for_deletion;
88  }
89 
90  private:
91  ComponentIndex componentDataIndex_[kMaxComponentCount];
92  bool marked_for_deletion_;
93 };
94 /// @}
95 
96 } // corgi
