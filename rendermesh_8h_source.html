63  /// @brief The move constructor for RenderMeshData to avoid reallocating
64  /// shader transforms.
65  ///

70  /// @brief The move assignment operator for RenderMeshData to avoid
71  /// reallocating shader transforms.
72  ///
73  /// @param[in] other The other RenderMeshData whose data should be moved
74  /// into this RenderMeshData.
75  ///
76  /// @return Returns a reference to the RenderMeshData that received the
77  /// data from `other`.
78  RenderMeshData& operator=(RenderMeshData&& other) {
79  mesh = std::move(other.mesh);
80  shader = std::move(other.shader);
81  tint = std::move(other.tint);
82  mesh_filename = std::move(other.mesh_filename);
83  shader_filename = std::move(other.shader_filename);
84  z_depth = std::move(other.z_depth);
85  culling_mask = std::move(other.culling_mask);
86  pass_mask = std::move(other.pass_mask);
87  visible = std::move(other.visible);
88  default_pose = std::move(other.default_pose);
89  num_shader_transforms = other.num_shader_transforms;
90  shader_transforms = other.shader_transforms;
91  other.shader_transforms = nullptr;
92  other.num_shader_transforms = 0;
93  return *this;
94  }

144 /// @brief Struct used for keeping track of and sorting our render lists.
145 struct RenderlistEntry {
146  /// @brief Constructor for RenderlistEntry.
147  RenderlistEntry(EntityRef entity_, RenderMeshData* data_)
148  : entity(entity_), data(data_) {}
149 
150  /// @brief The Entity associated with this RenderlistEntry.
151  EntityRef entity;
152 
153  /// @brief The RenderMeshData associated with the `entity` in this
154  /// RenderlistEntry.
155  RenderMeshData* data;
156 
157  /// @brief The greater than operator for RenderlistEntry.
158  ///
159  /// @param[in] other The other RenderlistEntry to compare if its
160  /// `RenderMeshData->z_depth` is greater than this RenderlistEntry's.
161  ///
162  /// @return Returns `true` if `other`'s `RenderMeshData->z_depth` is greater
163  /// than this RenderlistEntry's. Otherwise, it returns `false`.
164  bool operator<(const RenderlistEntry& other) const {
165  return (data->z_depth < other.data->z_depth);
166  }
167 
168  /// @brief The less than operator for RenderlistEntry.
169  ///
170  /// @param[in] other The other RenderlistEntry to compare if its
171  /// `RenderMeshData->z_depth` is less than this RenderlistEntry's.
172  ///
173  /// @return Returns `true` if `other`'s `RenderMeshData->z_depth` is less
174  /// than this RenderlistEntry's. Otherwise, it returns `false`.
175  bool operator>(const RenderlistEntry& other) const {
176  return (data->z_depth > other.data->z_depth);
177  }
178 };
