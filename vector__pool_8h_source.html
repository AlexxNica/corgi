62  /// @class VectorPoolReference
63  ///

520  /// @brief Get a VectorPoolReference to a new element.
521  ///
522  /// @note This function grabs the first free element (if one exists).
523  /// Otherwise, it allocates a new one on the underlying vector.
524  ///
525  /// @param[in] alloc_location An AllocationLocation enum determining whether
526  /// to add to the front or back of the underlying vector.
527  ///
528  /// @return Returns a VectorPoolReference pointing to the new element.
529  VectorPoolReference GetNewElement(AllocationLocation alloc_location) {
530  size_t index;
531  if (elements_[kFirstFree].next != kLastFree) {
532  index = elements_[kFirstFree].next;
533  RemoveFromList(index); // remove it from the list of free elements.
534  } else {
535  index = elements_.size();
536  elements_.push_back(VectorPoolElement());
537  }
538  switch (alloc_location) {
539  case kAddToFront:
540  AddToListFront(index, kFirstUsed);
541  break;
542  case kAddToBack:
543  AddToListBack(index, kLastUsed);
544  break;
545  default:
546  assert(0);
547  }
548  active_count_++;
549  // Placement new, to make sure we always give back a cleanly constructed
550  // element:
551  elements_[index].data.~T();
552  new (&(elements_[index].data)) T;
553  elements_[index].unique_id = AllocateUniqueId();
554  return VectorPoolReference(this, index);
555  }

557  /// @brief Frees an element at a given index.
558  ///
559  /// @note This removes the element from the list of active elements, and
560  /// adds it to the front of the inactive list (to be used later, when we
561  /// add elements to the VectorPool).
562  ///
563  /// @param[in] index The index corresponding to the element that should be
564  /// freed.
565  void FreeElement(size_t index) {
566  assert(elements_[index].unique_id != kInvalidId);
567  // Don't call the destructor directly - instead, assign over it.
568  // This ensures that data will always contain an object that is safe to
569  // destruct.
570  elements_[index].data.~T();
571  new (&(elements_[index].data)) T;
572  elements_[index].unique_id = kInvalidId;
573  RemoveFromList(index);
574  AddToListFront(index, kFirstFree);
575  active_count_--;
576  }

787  next_unique_id_++;

789  return result;
